Example script to post form data to the API endpoint

The form data is passed in as a JSON object

javascript

var formData = {
    "name": "John Doe",
    "email": "example@savantly.net",
    "message": "This is a test message"
};

// The API endpoint URL is passed in as a string
fetch('{{ENDPOINT}}', {
    method: 'POST',
    headers: {
        'Content-Type': 'application/json',
    },
    body: JSON.stringify(formData),
})
.then(response => response.json())
.then(data => {
    console.log('Success:', data);
})
.catch((error) => {
    console.error('Error:', error);
});
